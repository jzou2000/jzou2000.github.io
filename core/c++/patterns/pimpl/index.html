Pitfalls

  • Explicit destructor should be implemented in the source file where Pimpl is defined and implemented, otherwise the compiler fails to delete Pimpl object. For example 
    // in file a.h
class A {
    A();
    ~A() {}
private:
    class Pimpl;
    //Pimpl* u;
    std::unique_ptr<Pimpl> u;
};
    and 
    class A::Pimpl {
    friend class A;
    Pimpl() ...
    // unless using unique_ptr, ~Pimpl should be public (or accessible by owner that deletes it)
    ~Pimpl() ...
};
A::A() : u(new A::Pimpl()) {}
// A::~A() should be implemented here (even it is empty), when A::Pimpl is deleted
