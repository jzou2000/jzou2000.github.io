choice = [
    win: ['w10', 'w11', 'w2016'],
    dm:  ['unixodbc', 'iodbc'],
    'dm-osx': ['2.3.11', '3.52.8'],
]
random = new Random()
def pick(set, n=null) {
    def sz = set ? set.size() : 0
    return sz ? set[n !== null ? n : Math.abs(random.nextInt())%sz] : null
}
for (key in ['win', 'dm', 'dmc']) {
    values = choice[key]
    println("${key}: ${values}")
    if (values) {
        print '     '
        for (n = 0; n < 5; ++n) {
        	print(" ${pick(values)}")
        }
        println()
    }
}
