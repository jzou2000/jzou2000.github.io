markdonw-it + highlight.js

markdown-it - modern pluggable markdown parser (1.5M).

highlight.js - syntax highlighting with language autodetection. (2.4M) (not highlightjs, which is obsolete already)

/*
Demo: usage of markdown-it and highlight.js

*/
const log = console.log
const hljs = require('highlight')
const md = require('markdown-it')({
  breaks: true,
  highlight: function (str, lang) {
    if (lang && hljs.getLanguage(lang)) {
      try {
        return hljs.highlight(lang, str).value
      } catch (__) {}
    }
    return ''
  }
})
