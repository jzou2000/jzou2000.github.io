test plan (XM)

  1. import Branch.settings
  2. import DriverManager.settings
  3. PreTest
    1. remove staging
    2. install driver at staging/drv/ libXXX.so <== driver.zip DBF/ <== $(NFS)/SEN/SimbaEngine/$(SEN_BRANCH)/Multi-platform/DBF/XMDriver/DBF.zip ErrorMessages/ <== driver.zip
    3. prepare test cases at staging/test/
      1. NFS_TOUCHSTONE/TestCases.zip
      2. NFS/Drivers/DriverTest/DriverTest.zip
      3. driver.zip/Touchstone/specific
    4. win32: setup registry
    5. posix: setup ini
      • staging/odbc.ini
      • staging/odbcinst.ini
      • staging/simba.$(PRODUCT).ini
    6. export env
      • ODBCINI=path/odbc.ini
      • ODBCINSTINI=path/odbcinst.ini
      • ODBCSYSINI=path
      • SIMBAINI=path/simba.xm.ini
      • LD_LIBRARY_PATH=$(DM)/lib:$(STAGING)/lib:$(DRV):/opt/gcc/4.9.2/lib:/opt/gcc/4.9.2/lib64
      • DYLD_LIBRARY_PATY/LIBPATH
  4. TouchstoneTestList
    • dm_encoding=$(DM_ENCODING)
    • touchstone=ts binary
    • Locator=$(DS_LOCATOR)
    • TestList=$(TESTSUITE_LIST)
    • DataSourceVersion=default
  5. copy staging/test_output to $(BAMBOO_WORKING_DIRECTORY) for junit’s parsing
