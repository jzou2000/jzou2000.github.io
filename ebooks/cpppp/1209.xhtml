{
private:
    char * str;    // points to string allocated by new

};

  1. What’s wrong with this default constructor?

    String::String() {}

  2. What’s wrong with this constructor?

    String::String(const char * s)
{
    str = s;

}

  3. What’s wrong with this constructor?

    String::String(const char * s)
